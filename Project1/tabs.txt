CREATE TABLE EMPLOYEE(F_CPERSNO VARCHAR(15) NOT NULL, F_VC2NAME VARCHAR(50) NOT NULL, F_VC2DESIGNATION VARCHAR(15) NOT NULL, F_VC2DIVISION VARCHAR(15) NOT NULL, PRIMARY KEY(F_CPERSNO));

CREATE TABLE PASSWORDS(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, USERID VARCHAR(15) NOT NULL, PASSWORD VARCHAR(15) NOT NULL, PRIMARY KEY(USERID,PASSWORD,F_CPERSNO));

CREATE TABLE MPR_REPORTS(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, REP_DATE DATE NOT NULL, REP_NO VARCHAR(100) NOT NULL, TITLE VARCHAR(200) NOT NULL, PRIMARY KEY(F_CPERSNO, REP_DATE, REP_NO));

CREATE TABLE MPR_PAPERS(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, PAPER_DATE DATE NOT NULL, TITLE VARCHAR(200) NOT NULL, JOUR_NAME VARCHAR(200) NOT NULL, PRIMARY KEY(F_CPERSNO, PAPER_DATE, JOUR_NAME));

CREATE TABLE MPR_CEP(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, FROM_DATE DATE NOT NULL, TO_DATE DATE NOT NULL, TITLE VARCHAR(200) NOT NULL, PRIMARY KEY(F_CPERSNO, FROM_DATE, TITLE));

CREATE TABLE MPR_TC(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, TC_DATE DATE NOT NULL, TITLE VARCHAR(200) NOT NULL, PRIMARY KEY(F_CPERSNO, TC_DATE, TITLE));

CREATE TABLE MPR_TD(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, FROM_DATE DATE NOT NULL, TO_DATE DATE NOT NULL, PLACE VARCHAR(200) NOT NULL, PURPOSE VARCHAR(200) NOT NULL, PRIMARY KEY(F_CPERSNO, FROM_DATE, PLACE, PURPOSE));

CREATE TABLE MPR_LEAVES(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, FROM_DATE DATE NOT NULL, TO_DATE DATE NOT NULL, LEAVE_TYPE VARCHAR(100) NOT NULL, PRIMARY KEY(F_CPERSNO, FROM_DATE, LEAVE_TYPE));

CREATE TABLE MPR_FABRICATIONVISITS(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, VISIT_DATE DATE NOT NULL, INDUSTRY VARCHAR(100) NOT NULL, PURPOSE VARCHAR(500), PRIMARY KEY(F_CPERSNO, VISIT_DATE, PURPOSE));

CREATE TABLE MPR_MEETINGS(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, MEETING_DATE DATE NOT NULL, PROJECT_NAME VARCHAR(50) NOT NULL, PURPOSE VARCHAR(1000) NOT NULL, PRIMARY KEY(F_CPERSNO, MEETING_DATE, PROJECT_NAME));

CREATE TABLE MPR_ALL(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, FDATE DATE, INPUT_DATA VARCHAR(1000), INPUT_FLAG VARCHAR(20), PRIMARY KEY(F_CPERSNO, FDATE, INPUT_DATA, INPUT_FLAG));

CREATE TABLE PROJECT(PROJECT_NAME VARCHAR(30) PRIMARY KEY, DESCRIPTION VARCHAR(100));

CREATE TABLE MPR_WORKPROGRESS(F_CPERSNO VARCHAR(15) REFERENCES EMPLOYEE(F_CPERSNO) ON DELETE CASCADE, WP_DATE DATE NOT NULL, PROJECT_NAME VARCHAR(15) REFERENCES PROJECT(PROJECT_NAME) NOT NULL, DESCRIPTION VARCHAR(500), STATUS VARCHAR(500) NOT NULL, PDC DATE NOT NULL, PRIMARY KEY(F_CPERSNO, PROJECT_NAME, PDC, WP_DATE));
